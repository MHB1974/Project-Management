import { useState, useRef, useEffect } from 'react';
import './ClaudeChat.css';

export default function ClaudeChat({ profile, isAdmin }) {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');
  const [loading, setLoading] = useState(false);
  const [apiKeySet, setApiKeySet] = useState(false);
  const [showApiKeyInput, setShowApiKeyInput] = useState(false);
  const messagesEndRef = useRef(null);

  useEffect(() => {
    const savedKey = localStorage.getItem('claude-api-key');
    setApiKeySet(!!savedKey);
  }, []);

  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);

  const getApiKey = () => {
    return localStorage.getItem('claude-api-key') || '';
  };

  const saveApiKey = (key) => {
    if (key.trim()) {
      localStorage.setItem('claude-api-key', key.trim());
      setApiKeySet(true);
      setShowApiKeyInput(false);
    }
  };

  const sendMessage = async () => {
    if (!input.trim()) return;
    if (!apiKeySet) {
      alert('먼저 Claude API 키를 설정하세요.');
      setShowApiKeyInput(true);
      return;
    }

    const userMessage = input.trim();
    setInput('');
    setMessages(prev => [...prev, { role: 'user', text: userMessage }]);
    setLoading(true);

    try {
      const apiKey = getApiKey();
      const response = await fetch('https://api.anthropic.com/v1/messages', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'x-api-key': apiKey,
          'anthropic-version': '2023-06-01',
        },
        body: JSON.stringify({
          model: 'claude-opus-4-6',
          max_tokens: 1024,
          messages: [
            ...messages.map(m => ({
              role: m.role,
              content: m.text,
            })),
            { role: 'user', content: userMessage },
          ],
        }),
      });

      if (!response.ok) {
        const error = await response.json();
        throw new Error(error.error?.message || 'API 오류');
      }

      const data = await response.json();
      const assistantMessage = data.content[0]?.text || '답변 생성 실패';
      
      setMessages(prev => [...prev, { role: 'assistant', text: assistantMessage }]);
    } catch (error) {
      setMessages(prev => [...prev, {
        role: 'assistant',
        text: `오류: ${error.message}`,
        isError: true,
      }]);
    } finally {
      setLoading(false);
    }
  };

  if (!isAdmin) {
    return null;
  }

  return (
    <div className="claude-chat">
      <div className="claude-header">
        <div className="claude-title">
          💬 Claude (관리자용)
        </div>
        <button
          className="claude-settings"
          onClick={() => setShowApiKeyInput(!showApiKeyInput)}
          title="API 키 설정"
        >
          ⚙️
        </button>
      </div>

      {showApiKeyInput && (
        <ApiKeyInput onSave={saveApiKey} />
      )}

      {!apiKeySet && !showApiKeyInput && (
        <div className="claude-setup">
          <div style={{ fontSize: 12, color: 'var(--muted)' }}>
            Claude API 키를 설정하세요
          </div>
          <button
            className="btn sm"
            onClick={() => setShowApiKeyInput(true)}
            style={{ marginTop: 8, width: '100%' }}
          >
            API 키 설정
          </button>
          <div style={{ fontSize: 11, color: 'var(--muted)', marginTop: 8, lineHeight: 1.4 }}>
            <a href="https://console.anthropic.com" target="_blank" rel="noopener noreferrer"
               style={{ color: 'var(--accent)' }}>
              console.anthropic.com
            </a>
            {' '}에서 발급받으세요
          </div>
        </div>
      )}

      {apiKeySet && !showApiKeyInput && (
        <>
          <div className="claude-messages">
            {messages.length === 0 && (
              <div className="claude-empty">
                프로젝트 관리 및 분석에 대해 물어보세요! 📊
              </div>
            )}
            {messages.map((msg, idx) => (
              <div key={idx} className={`claude-message ${msg.role}`}>
                <div className={`message-bubble ${msg.isError ? 'error' : ''}`}>
                  {msg.text}
                </div>
              </div>
            ))}
            {loading && (
              <div className="claude-message assistant">
                <div className="message-bubble typing">
                  <span></span><span></span><span></span>
                </div>
              </div>
            )}
            <div ref={messagesEndRef} />
          </div>

          <div className="claude-input-area">
            <textarea
              className="claude-input"
              value={input}
              onChange={e => setInput(e.target.value)}
              onKeyPress={e => {
                if (e.key === 'Enter' && !e.shiftKey) {
                  e.preventDefault();
                  sendMessage();
                }
              }}
              placeholder="메시지를 입력하세요... (Shift+Enter로 줄바꿈)"
              disabled={loading}
              rows={2}
            />
            <button
              className="claude-send"
              onClick={sendMessage}
              disabled={loading || !input.trim()}
              title="전송 (Enter)"
            >
              📤
            </button>
          </div>
        </>
      )}
    </div>
  );
}

function ApiKeyInput({ onSave }) {
  const [key, setKey] = useState('');
  const [showKey, setShowKey] = useState(false);

  return (
    <div className="api-key-input">
      <div style={{ fontSize: 12, marginBottom: 8 }}>
        API 키 설정
      </div>
      <div style={{ display: 'flex', gap: 8, marginBottom: 8 }}>
        <input
          type={showKey ? 'text' : 'password'}
          value={key}
          onChange={e => setKey(e.target.value)}
          placeholder="sk-ant-..."
          className="field"
          style={{ flex: 1 }}
        />
        <button
          className="btn ghost sm"
          onClick={() => setShowKey(!showKey)}
          style={{ padding: '6px 10px' }}
          title="표시/숨김"
        >
          {showKey ? '숨김' : '표시'}
        </button>
      </div>
      <div style={{ display: 'flex', gap: 6 }}>
        <button
          className="btn sm"
          onClick={() => onSave(key)}
          disabled={!key.trim()}
          style={{ flex: 1 }}
        >
          저장
        </button>
      </div>
      <div style={{ fontSize: 11, color: 'var(--muted)', marginTop: 8, lineHeight: 1.4 }}>
        API 키는 로컬 스토리지에만 저장되며, 서버로 전송되지 않습니다.
      </div>
    </div>
  );
}
